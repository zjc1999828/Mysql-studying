```sql
SELECT employee.empno AS 员工编号,
employee.ename AS 员工名字, 
employee.mgr AS 员工上级编号,
worker.empno AS 上级编号,
worker.ename AS 上级名字 
FROM emp worker ,emp employee 
WHERE employee.mgr = worker.empno
```
